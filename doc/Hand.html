def is_blackjack
  face = false
  ace = false
  @hand_cards.each { |card|
    if card.value == 10
      face = true
    end
    if card.value == 1
      ace = true
    end
  }
  if face && ace && (@hand_cards.size == 2)
    return true
  end
  return false
end

def update_points
  @points = 0
  ace_present = false
  @hand_cards.each { |card|
    @points = @points + card.value
    if card.value == 1
      ace_present = true
    end
  }
  if ace_present && (@points + 10 <= 21)
    @points = @points + 10
  end
end
