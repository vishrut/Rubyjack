    Make a bet of the given value

    # File Player.rb, line 112
def make_bet(bet_value)
  if bet_value > @money
    return false
  end
  
  if bet_value <= 0
    return false
  end
  
  @current_bet = bet_value
  @money = @money - @current_bet
  return true
end
