def reset
  for i in 1..@size
    deck = Deck.new
    @cards.concat(deck.get_cards)
  end
  @cards = @cards.shuffle
end

def retrieve_card
  card = @cards.pop
  if card.nil?
    self.reset_shoe
    card = @cards.pop
  end
  return card
end

    Initialize a shoe with the size number of decks

def initialize(size)
  @size = size
  @cards = Array.new
  for i in 1..@size
    deck = Deck.new
    @cards.concat(deck.get_cards)
  end
  
  # Shuffle all the cards in the deck
  @cards = @cards.shuffle
end
